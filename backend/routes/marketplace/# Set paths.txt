# Set paths
$mdFolder = "C:\CFH\docs\features"           # Folder containing .md files
$mainProjectFolder = "C:\CFH"                # Root folder of your codebase
$outputFile = "C:\CFH\docs\features\file_reference_report.txt"

# File extensions to search for in .md files
$fileExtensions = @("ts", "js", "json", "tsx", "css", "scss")

# Build hashtable for fast lookup of all project files
Write-Host "📂 Indexing all project files in $mainProjectFolder..."
$allProjectFiles = @{}
Get-ChildItem -Path $mainProjectFolder -Recurse -File -ErrorAction SilentlyContinue | ForEach-Object {
    $relativePath = $_.FullName.Substring($mainProjectFolder.Length).TrimStart("\") -replace "/", "\"
    $allProjectFiles[$relativePath.ToLower()] = $true
}
Write-Host "✅ Indexed $($allProjectFiles.Count) files."

# Store results
$results = @()

# Process each .md file
Write-Host "🔍 Scanning .md files in $mdFolder..."
Get-ChildItem -Path $mdFolder -Filter *.md | ForEach-Object {
    $mdFile = $_
    $content = Get-Content $mdFile.FullName -Raw

    foreach ($ext in $fileExtensions) {
        # Regex to match filenames like some/path/file.ts
        $regex = "\b[\w\-/\\\.]+\." + [Regex]::Escape($ext)
        $matches = [regex]::Matches($content, $regex) | Select-Object -Unique

        foreach ($match in $matches) {
            # Normalize slashes and trim
            $fileReference = ($match.Value -replace "/", "\" -replace "\\+", "\").Trim()
            $relativeRef = $fileReference.TrimStart("\").ToLower()

            # Check if file exists using hashtable
            $exists = $allProjectFiles.ContainsKey($relativeRef)

            # Add to results
            $results += [PSCustomObject]@{
                MarkdownFile   = $mdFile.Name
                ReferencedFile = $fileReference
                ExistsInCFH    = if ($exists) { "Yes" } else { "No" }
            }
        }
    }
}

# Sort and format output
$results = $results | Sort-Object MarkdownFile, ReferencedFile | Select-Object MarkdownFile, ReferencedFile, ExistsInCFH

# Output results to file
$results | Format-Table -AutoSize | Out-File -FilePath $outputFile -Encoding UTF8

# Summary
$missingCount = ($results | Where-Object { $_.ExistsInCFH -eq "No" }).Count
$totalCount = $results.Count

Write-Host "`n✅ Report generated at: $outputFile"
Write-Host "📊 Total references found: $totalCount"
Write-Host "❌ Missing files: $missingCount"
