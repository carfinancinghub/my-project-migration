# Step 1: Define the exact files you want to find
$targetFiles = @(
	"TintingServiceDiscovery.tsx"
	
)

# Step 2: Set root path and exclusion for node_modules
$basePath = "C:\CFH"
$excludeRegex = "node_modules"

# Step 3: Search and extract file data
$results = Get-ChildItem -Path $basePath -Recurse -File -ErrorAction SilentlyContinue |
    Where-Object {
        $_.FullName -notmatch $excludeRegex -and
        $_.Name -in $targetFiles
    } |
    Select-Object @{
        Name = 'Name'; Expression = { $_.Name }
    }, @{
        Name = 'FullPath'; Expression = { $_.FullName }
    }, @{
        Name = 'LastWriteTime'; Expression = { $_.LastWriteTime }
    }, @{
        Name = 'Size(Bytes)'; Expression = { $_.Length }
    } |
    Sort-Object Name

# Step 4: Output each result on a single detailed line
$results | ForEach-Object {
    "{0,-35} | {1,-22} | {2,10} bytes | {3}" -f $_.Name, $_.'LastWriteTime'.ToString("yyyy-MM-dd HH:mm:ss"), $_.'Size(Bytes)', $_.FullPath
}
