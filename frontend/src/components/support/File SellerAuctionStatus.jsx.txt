import React, { useEffect, useState } from 'react';
import PropTypes from 'prop-types';
import logger from '@utils/logger';
import { generateAICompetitiveBidInsights } from '@services/ai/BidAnalytics';
import { trackAuctionGamificationMilestones } from '@services/gamification/MilestoneTracker';

/**
 * Functions Summary:
 * - generateAICompetitiveBidInsights(auctionId): Retrieves competitive bid insights using AI.
 * - trackAuctionGamificationMilestones(auctionId): Tracks gamification status for sellers.
 * Inputs: message (string), auctionId (string)
 * Outputs: JSX with message and optional AI insights
 * Dependencies: React, @services/ai/BidAnalytics, @services/gamification/MilestoneTracker, @utils/logger
 */
const SellerAuctionStatus = ({ message, auctionId }) => {
  const [insights, setInsights] = useState([]);
  const [insightError, setInsightError] = useState(null);

  useEffect(() => {
    if (!auctionId) return;

    const fetchInsights = async () => {
      try {
        const data = await generateAICompetitiveBidInsights(auctionId);
        setInsights(data);
      } catch (error) {
        logger.error('Failed to generate AI bid insights:', error);
        setInsightError('Failed to load bid insights');
      }
    };

    const trackGamification = () => {
      try {
        trackAuctionGamificationMilestones(auctionId);
      } catch (error) {
        logger.error('Gamification milestone tracking failed:', error);
      }
    };

    fetchInsights();
    trackGamification();
  }, [auctionId]);

  try {
    if (!message) throw new Error('Message prop is required');

    return (
      <div className="p-4 bg-white rounded shadow">
        <div className="text-lg font-semibold">{message}</div>
        {insightError ? (
          <div className="text-red-600 mt-2">{insightError}</div>
        ) : (
          <ul className="mt-2 space-y-2">
            {insights.map((item, idx) => (
              <li key={idx} className="text-gray-700">{item.summary}</li>
            ))}
          </ul>
        )}
      </div>
    );
  } catch (error) {
    logger.error('SellerAuctionStatus error:', error);
    return <div>Error displaying message</div>;
  }
};

SellerAuctionStatus.propTypes = {
  message: PropTypes.string.isRequired,
  auctionId: PropTypes.string.isRequired,
};

export default SellerAuctionStatus;

import React, { useState } from 'react';
import PropTypes from 'prop-types';
import logger from '@utils/logger';
import FileUploader from '@components/common/FileUploader';
import LoyaltyBoostChecker from '@components/common/LoyaltyBoostChecker';

/**
 * Functions Summary:
 * - handleSubmit(): Handles ticket submission, logs error if failed
 * - onFileUpload(): Sets local file state
 * Inputs: isPremium, userId
 * Outputs: JSX UI for support ticket entry
 * Dependencies: logger, FileUploader, LoyaltyBoostChecker
 */
const SupportTicketForm = ({ isPremium, userId }) => {
  const [subject, setSubject] = useState('');
  const [description, setDescription] = useState('');
  const [file, setFile] = useState(null);
  const [priorityBoost, setPriorityBoost] = useState(false);
  const [message, setMessage] = useState(null);

  const handleSubmit = (e) => {
    e.preventDefault();
    try {
      const payload = {
        userId,
        subject,
        description,
        file: file?.name,
        priorityBoost: isPremium ? priorityBoost : false,
      };
      logger.info('Support ticket submitted:', payload);
      setMessage('Ticket submitted successfully!');
    } catch (error) {
      logger.error('Support ticket submission failed:', error);
      setMessage('Failed to submit ticket.');
    }
  };

  return (
    <form onSubmit={handleSubmit} className="p-4 bg-white rounded shadow">
      <h3 className="text-xl font-bold mb-2">Submit a Support Ticket</h3>

      <input
        type="text"
        placeholder="Subject"
        value={subject}
        onChange={(e) => setSubject(e.target.value)}
        className="border p-2 mb-2 w-full"
        required
      />

      <textarea
        placeholder="Describe your issue"
        value={description}
        onChange={(e) => setDescription(e.target.value)}
        className="border p-2 mb-2 w-full"
        rows={4}
        required
      />

      {isPremium ? (
        <>
          <FileUploader onUpload={setFile} />
          <LoyaltyBoostChecker
            isPremium={isPremium}
            checked={priorityBoost}
            onChange={() => setPriorityBoost(!priorityBoost)}
          />
        </>
      ) : (
        <div className="text-yellow-600 italic mb-2">Premium support features unavailable. Upgrade to unlock file upload and priority boost.</div>
      )}

      <button type="submit" className="bg-blue-600 text-white px-4 py-2 rounded">
        Submit Ticket
      </button>

      {message && <p className="mt-3 text-sm">{message}</p>}
    </form>
  );
};

SupportTicketForm.propTypes = {
  isPremium: PropTypes.bool.isRequired,
  userId: PropTypes.string.isRequired,
};

export default SupportTicketForm;

import React, { useEffect, useState } from 'react';
import PropTypes from 'prop-types';
import logger from '@utils/logger';
import { runEquityOptimizedFinancingMatch } from '@services/ai/FinancingOptimizer';
import BlockchainTrailViewer from '@components/blockchain/BlockchainTrailViewer';
import ProgressBadgeChart from '@components/common/ProgressBadgeChart';

/**
 * Functions Summary:
 * - runEquityOptimizedFinancingMatch(auctionId): Calls AI module to fetch financing options.
 * - renderBlockchainFinancingTrail(id): Renders blockchain audit viewer for transparency.
 * - renderProgressChart(): Shows badge progress graph using mock badge levels.
 * Inputs: auctionId (string), isPremium (boolean)
 * Outputs: JSX displaying financing, progress, and blockchain data
 * Dependencies: FinancingOptimizer, BlockchainTrailViewer, ProgressBadgeChart
 */
const BuyerFinancingModal = ({ auctionId, isPremium }) => {
  const [financingOptions, setFinancingOptions] = useState([]);
  const [financingError, setFinancingError] = useState(null);
  const [trailError, setTrailError] = useState(null);
  const [recommendation, setRecommendation] = useState('');

  useEffect(() => {
    const fetchFinancing = async () => {
      try {
        const matches = await runEquityOptimizedFinancingMatch(auctionId);
        setFinancingOptions(matches);
        if (isPremium && matches.length > 0) {
          setRecommendation('Our AI recommends proceeding with the top lender for fastest approval.');
        }
      } catch (error) {
        logger.error('Failed to run equity-optimized financing match:', error);
        setFinancingError('Failed to load financing options');
      }
    };

    if (auctionId) fetchFinancing();
  }, [auctionId, isPremium]);

  const renderBlockchainFinancingTrail = (id) => {
    try {
      return <BlockchainTrailViewer trailId={id} />;
    } catch (error) {
      logger.error('Failed to render blockchain financing trail:', error);
      setTrailError('Unable to load blockchain trail.');
      return <div>Blockchain trail unavailable</div>;
    }
  };

  const renderProgressChart = () => {
    const mockProgress = { currentLevel: 3, totalLevels: 5, progressPercentage: 60 };
    return <ProgressBadgeChart data={mockProgress} />;
  };

  if (!auctionId) return <div>Invalid auction ID</div>;

  return (
    <div className="p-4 bg-white rounded shadow">
      <h2 className="text-xl font-bold mb-4">Financing Options</h2>
      {financingError ? (
        <div className="text-red-600">{financingError}</div>
      ) : financingOptions.length === 0 ? (
        <div>Loading options...</div>
      ) : (
        <>
          <ul className="space-y-4">
            {financingOptions.map((option, index) => (
              <li key={index}>
                <div>{option.lender} - {option.rate}% APR</div>
                {renderBlockchainFinancingTrail(option.blockchainId)}
              </li>
            ))}
          </ul>
          {isPremium && recommendation && (
            <div className="mt-4 text-green-700 font-medium">{recommendation}</div>
          )}
        </>
      )}
      {trailError && <div className="text-red-600 mt-2">{trailError}</div>}
      {isPremium && (
        <div className="mt-6">
          <h3 className="text-md font-semibold mb-2">Financing Badge Progress</h3>
          {renderProgressChart()}
        </div>
      )}
    </div>
  );
};

BuyerFinancingModal.propTypes = {
  auctionId: PropTypes.string.isRequired,
  isPremium: PropTypes.bool.isRequired,

        <div key={delivery.id} className="mb-4 border-b pb-2">
          <div><strong>Item:</strong> {delivery.item}</div>
          <div><strong>Pickup:</strong> {delivery.pickup}</div>
          <div><strong>Dropoff:</strong> {delivery.dropoff}</div>
          <div><strong>Date:</strong> {delivery.date}</div>
          <div><strong>Status:</strong> {status[delivery.id] || 'Pending'}</div>
          <button
            className="bg-green-600 text-white px-2 py-1 mt-2 rounded"
            onClick={() => markAsDelivered(delivery.id)}
          >
            Confirm Delivery
          </button>
          {isPremium && (
            <button
              className="ml-2 bg-indigo-600 text-white px-2 py-1 rounded"
              onClick={() => verifyPOD(delivery.id)}
            >
              Verify POD (Blockchain)
            </button>
          )}
          {verificationStatus[delivery.id] && (
            <div className="text-green-600 text-sm mt-1">
              Blockchain Verified
            </div>
          )}
        </div>
      ))}
      {routeInfo && (
        <div className="mt-4 p-2 border bg-gray-100 rounded">
          <h3 className="font-semibold">Optimized Route</h3>
          <pre className="text-sm">{routeInfo.optimizedPath}</pre>
        </div>
      )}
      {error && <div className="text-red-600 mt-2">{error}</div>}
      {!isPremium && (
        <div className="mt-4 text-yellow-600 italic">
          Premium features (route optimization, blockchain POD) are disabled. Upgrade to unlock.
        </div>
      )}
    </div>
  );
};

HaulerAuctionDelivery.propTypes = {
  deliveries: PropTypes.arrayOf(
    PropTypes.shape({
      id: PropTypes.string.isRequired,
      item: PropTypes.string.isRequired,
      pickup: PropTypes.string.isRequired,
      dropoff: PropTypes.string.isRequired,
      date: PropTypes.string.isRequired,
    })
  ).isRequired,
  isPremium: PropTypes.bool.isRequired,
};

// File: BuyerFinancingModal.jsx
// Path: frontend/src/components/buyer/BuyerFinancingModal.jsx
// Purpose: Buyer modal to display equity-optimized financing with blockchain audit trail and badge gamification
// Author: Cod1 (05101340 - PDT), Editor: Cod1 (05141445 - PDT) — Added badge progress graph and AI action advisor
// 👑 Cod2 Crown Certified

import React, { useEffect, useState } from 'react';
import PropTypes from 'prop-types';
import logger from '@utils/logger';
import { runEquityOptimizedFinancingMatch, suggestNextFinancingAction } from '@services/ai/FinancingOptimizer';
import BlockchainTrailViewer from '@components/blockchain/BlockchainTrailViewer';
import FinancingProgressGraph from '@components/gamification/FinancingProgressGraph';

/**
 * Functions Summary:
 * - runEquityOptimizedFinancingMatch(auctionId): Fetch financing options.
 * - suggestNextFinancingAction(auctionId): AI-suggested next move (premium).
 * - renderBlockchainFinancingTrail(id): Renders audit log (premium).
 * Inputs: auctionId (string), isPremium (bool)
 * Outputs: JSX with financing list, blockchain, gamification graph
 * Dependencies: @services/ai/FinancingOptimizer, BlockchainTrailViewer, FinancingProgressGraph
 */
const BuyerFinancingModal = ({ auctionId, isPremium }) => {
  const [financingOptions, setFinancingOptions] = useState([]);
  const [financingError, setFinancingError] = useState(null);
  const [trailError, setTrailError] = useState(null);
  const [nextMove, setNextMove] = useState('');
  const [graphLevel, setGraphLevel] = useState(0);

  useEffect(() => {
    const fetchFinancing = async () => {
      try {
        const matches = await runEquityOptimizedFinancingMatch(auctionId);
        setFinancingOptions(matches);
        if (isPremium) {
          const suggestion = await suggestNextFinancingAction(auctionId);
          setNextMove(suggestion);
          setGraphLevel(3); // Simulated
        }
      } catch (error) {
        logger.error('Financing fetch failed:', error);
        setFinancingError('Failed to load financing options');
      }
    };

    if (auctionId) fetchFinancing();
  }, [auctionId, isPremium]);

  const renderBlockchainFinancingTrail = (id) => {
    try {
      return <BlockchainTrailViewer trailId={id} />;
    } catch (error) {
      logger.error('Blockchain trail failed:', error);
      setTrailError('Unable to load blockchain trail.');
      return <div>Blockchain trail unavailable</div>;
    }
  };

  if (!auctionId) return <div>Invalid auction ID</div>;

  return (
    <div className="p-4 bg-white rounded shadow">
      <h2 className="text-xl font-bold mb-2">Financing Options</h2>
      {financingError ? (
        <div className="text-red-600">{financingError}</div>
      ) : financingOptions.length === 0 ? (
        <div>Loading options...</div>
      ) : (
        <ul className="space-y-4">
          {financingOptions.map((option, index) => (
            <li key={index}>
              <div>{option.lender} - {option.rate}% APR</div>
              {isPremium && renderBlockchainFinancingTrail(option.blockchainId)}
            </li>
          ))}
        </ul>
      )}
      {trailError && <div className="text-red-600 mt-2">{trailError}</div>}
      {isPremium && (
        <div className="mt-4">
          <h3 className="font-semibold">Badge Progress</h3>
          <FinancingProgressGraph level={graphLevel} />
          <div className="mt-2 text-blue-600">Suggested Move: {nextMove}</div>
        </div>
      )}
    </div>
  );
};

BuyerFinancingModal.propTypes = {
  auctionId: PropTypes.string.isRequired,
  isPremium: PropTypes.bool.isRequired,
};

export default BuyerFinancingModal;
